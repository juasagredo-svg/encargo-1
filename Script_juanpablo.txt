CREATE SEQUENCE seq_contrato

  START WITH 321

  INCREMENT BY 1

  NOCACHE;



CREATE SEQUENCE seq_categoria

  START WITH 1

  INCREMENT BY 1

  NOCACHE;





CREATE TABLE t_categoria_equipo (

  id_categoria    NUMBER    NOT NULL,

  nombre_categoria  VARCHAR2(50) NOT NULL,

  tarifa_diaria_base NUMBER(12,2) NOT NULL,

   

  CONSTRAINT pk_categoria_equipo PRIMARY KEY (id_categoria),

  CONSTRAINT uk_nombre_categoria UNIQUE (nombre_categoria)

);



CREATE TABLE t_cliente (

  rut_cliente    VARCHAR2(12)  NOT NULL,

  nombre_cliente  VARCHAR2(100) NOT NULL,

  domicilio     VARCHAR2(150) NOT NULL,

  email_cliente   VARCHAR2(100),

  telefono_cliente VARCHAR2(20),

   

  CONSTRAINT pk_cliente PRIMARY KEY (rut_cliente),

  CONSTRAINT uk_email_cliente UNIQUE (email_cliente)

);



CREATE TABLE t_empleado (

  id_empleado   NUMBER    NOT NULL,

  rut_empleado   VARCHAR2(12) NOT NULL,

  nombre_empleado VARCHAR2(100) NOT NULL,

  cargo      VARCHAR2(50) DEFAULT 'Gestor de Flota' NOT NULL,

   

  CONSTRAINT pk_empleado PRIMARY KEY (id_empleado),

  CONSTRAINT uk_rut_empleado UNIQUE (rut_empleado)

);



CREATE TABLE t_proveedor (

  id_proveedor   NUMBER    NOT NULL,

  rut_proveedor  VARCHAR2(12) NOT NULL,

  nombre_proveedor VARCHAR2(100) NOT NULL,

  telefono_proveedor VARCHAR2(15),

   

  CONSTRAINT pk_proveedor PRIMARY KEY (id_proveedor),

  CONSTRAINT uk_rut_proveedor UNIQUE (rut_proveedor)

);



CREATE TABLE t_equipo (

  codigo_equipo    VARCHAR2(10)  NOT NULL,

  nombre_equipo    VARCHAR2(100) NOT NULL,

  numero_serie    VARCHAR2(20)  NOT NULL,

  estado       VARCHAR2(20)  DEFAULT 'Disponible' NOT NULL,

  fk_id_categoria   NUMBER     NOT NULL,

  tarifa_especial_diaria NUMBER(12,2),

   

  CONSTRAINT pk_equipo PRIMARY KEY (codigo_equipo),

  CONSTRAINT uk_numero_serie UNIQUE (numero_serie),

  CONSTRAINT fk_equipo_categoria FOREIGN KEY (fk_id_categoria)

    REFERENCES t_categoria_equipo(id_categoria),

  CONSTRAINT ck_estado_equipo CHECK (estado IN ('Disponible', 'Mantenimiento', 'Arrendado', 'Reparación'))

);



CREATE TABLE t_contrato_arriendo (

  numero_contrato NUMBER    NOT NULL,

  fk_rut_cliente  VARCHAR2(12) NOT NULL,

  fk_id_empleado  NUMBER    NOT NULL,

  fecha_inicio   DATE     NOT NULL,

  fecha_termino  DATE     NOT NULL,

  estado_contrato VARCHAR2(20) DEFAULT 'Activo' NOT NULL,

   

  CONSTRAINT pk_contrato_arriendo PRIMARY KEY (numero_contrato),

  CONSTRAINT fk_contrato_cliente FOREIGN KEY (fk_rut_cliente)

    REFERENCES t_cliente(rut_cliente),

  CONSTRAINT fk_contrato_empleado FOREIGN KEY (fk_id_empleado)

    REFERENCES t_empleado(id_empleado),

  CONSTRAINT ck_estado_contrato CHECK (estado_contrato IN ('Activo', 'Finalizado', 'Cancelado'))

);



CREATE TABLE t_detalle_arriendo (

  fk_numero_contrato  NUMBER    NOT NULL,

  fk_codigo_equipo   VARCHAR2(10) NOT NULL,

  cantidad_arrendada  NUMBER    DEFAULT 1 NOT NULL,

  precio_diario_aplicado NUMBER(12,2) NOT NULL,

   

  CONSTRAINT pk_detalle_arriendo PRIMARY KEY (fk_numero_contrato, fk_codigo_equipo),

  CONSTRAINT fk_detalle_contrato FOREIGN KEY (fk_numero_contrato)

    REFERENCES t_contrato_arriendo(numero_contrato),

  CONSTRAINT fk_detalle_equipo FOREIGN KEY (fk_codigo_equipo)

    REFERENCES t_equipo(codigo_equipo),

  CONSTRAINT ck_cantidad_positiva CHECK (cantidad_arrendada > 0)

);



CREATE TABLE t_mantenimiento (

  id_mantenimiento  NUMBER    NOT NULL,

  fk_codigo_equipo  VARCHAR2(10) NOT NULL,

  fk_id_proveedor  NUMBER    NOT NULL,

  fecha_inicio    DATE     NOT NULL,

  fecha_fin     DATE,

  tipo_mantenimiento VARCHAR2(30) NOT NULL,

  costo_mantenimiento NUMBER(12,2),

  descripcion    VARCHAR2(500),

   

  CONSTRAINT pk_mantenimiento PRIMARY KEY (id_mantenimiento),

  CONSTRAINT fk_mantenimiento_equipo FOREIGN KEY (fk_codigo_equipo)

    REFERENCES t_equipo(codigo_equipo),

  CONSTRAINT fk_mantenimiento_proveedor FOREIGN KEY (fk_id_proveedor)

    REFERENCES t_proveedor(id_proveedor),

  CONSTRAINT ck_tipo_mantenimiento CHECK (tipo_mantenimiento IN ('Preventivo', 'Correctivo', 'Predictivo'))

);



CREATE TABLE t_pago (

  id_pago      NUMBER    NOT NULL,

  fk_numero_contrato NUMBER    NOT NULL,

  fecha_pago     DATE     DEFAULT SYSDATE NOT NULL,

  monto_pagado    NUMBER(12,2) NOT NULL,

  metodo_pago    VARCHAR2(20) NOT NULL,

  banco       VARCHAR2(50),

   

  CONSTRAINT pk_pago PRIMARY KEY (id_pago),

  CONSTRAINT fk_pago_contrato FOREIGN KEY (fk_numero_contrato)

    REFERENCES t_contrato_arriendo(numero_contrato),

  CONSTRAINT ck_metodo_pago CHECK (metodo_pago IN ('Tarjetaw', 'Transferencia', 'Efectivo'))

);

  START WITH 102

  INCREMENT BY 1

  NOCACHE;



CREATE OR REPLACE TRIGGER trg_empleado_automatico

BEFORE INSERT ON t_empleado

FOR EACH ROW

BEGIN

  IF :NEW.id_empleado IS NULL THEN

    SELECT seq_empleado.NEXTVAL INTO :NEW.id_empleado FROM DUAL;

  END IF;

END;

SELECT 

  e.codigo_equipo AS "Código Equipo",

  e.nombre_equipo AS "Nombre del Equipo",

  NVL(SUM(da.cantidad_arrendada), 0) AS "Total Veces Arrendado"

FROM t_equipo e

LEFT JOIN t_detalle_arriendo da ON e.codigo_equipo = da.fk_codigo_equipo

GROUP BY e.codigo_equipo, e.nombre_equipo

ORDER BY "Total Veces Arrendado" DESC, e.codigo_equipo;



SELECT 

  ca.numero_contrato AS "Número Contratoa",

  c.nombre_cliente AS "Cliente",

  SUM(da.cantidad_arrendada * da.precio_diario_aplicado) AS "Monto Total del Contrato"

FROM t_contrato_arriendo ca

JOIN t_cliente c ON ca.fk_rut_cliente = c.rut_cliente

JOIN t_detalle_arriendo da ON ca.numero_contrato = da.fk_numero_contrato

GROUP BY ca.numero_contrato, c.nombre_cliente

ORDER BY ca.numero_contrato;
